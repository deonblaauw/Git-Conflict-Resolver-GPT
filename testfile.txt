import random

def generate_random_numbers(n):
    random_numbers = [random.randint(1, 100) for _ in range(n)]
    return random_numbers

if __name__ == "__main__":
    n = 102
    random_numbers12 = generate_random_numbers(n)
    total_sum = sum(random_numbers12)
    print(f"The random numbers are: {random_numbers12}")
    print(f"The sum of the random numbers is: {total_sum}")
