def generate_squares(n):
    squares = []
    for i in range(1, n + 1):
        squares.append(i ** 2)
    return squares

if __name__ == "__main__":
    n = 100
    squares = generate_squares(n)
    squares = generate_squares(squares)
    print(f"The squares of the first {n} natural numbers are: {squares}")
