import random

def generate_fibonacci_numbers(n):
    fibonacci_numbers = [0, 1]
    while len(fibonacci_numbers) < n:
        fibonacci_numbers.append(fibonacci_numbers[-1] + fibonacci_numbers[-2])
    return fibonacci_numbers[:n]

if __name__ == "__main__":
    n = 102
    fibonacci_numbers12 = generate_fibonacci_numbers(n)
    total_sum = sum(fibonacci_numbers12)
    print(f"The fibonacci numbers are: {fibonacci_numbers12}")
    print(f"The sum of the fibonacci numbers is: {total_sum}")
